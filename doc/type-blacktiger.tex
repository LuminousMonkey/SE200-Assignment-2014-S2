\enableregime[utf]

\definefontfeature [concourse] [kern=yes]

\starttypescript [sans] [concourse]
  \setups[font:fallback:sans]
  \definefontsynonym [Sans] [Concourse-Regular] [features=default]
  \definefontsynonym [SansItalic] [Concourse-Italic] [features=default]
  \definefontsynonym [SansBold] [Concourse-Bold] [features=default]
  \definefontsynonym [SansBoldItalic] [Concourse-Bold Italic] [features=default]
\stoptypescript

\starttypescript [sans] [concourse]
  \definefontsynonym [Concourse-Regular] [name:Concourse T6 Regular] [features=default]
  \definefontsynonym [Concourse-Italic] [name:Concourse T6 Italic] [features=default]
  \definefontsynonym [Concourse-Bold] [name:Concourse T6 Bold] [features=default]
  \definefontsynonym [Concourse-BoldItalic] [name:Concourse T6 Bold Italic] [features=default]
\stoptypescript
